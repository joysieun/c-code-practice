#include <iostream>

int main(void)
{
	int total;
	double pay;
	
	while (true) {
		std::cout << "판매 금액을 만원 단위로 입력(-1 to end)";
		std::cin >> total;
		
		if (total == -1) break;
		pay = total * 0.12 + 50;
		std::cout << "이번 달 급여: " << pay << "만원" << std::endl;
		
		
	}
	std::cout << "판매가 종료되었습니다." << std::endl;

}
