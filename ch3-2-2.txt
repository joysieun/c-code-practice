#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <cstring>

class Printer {
private:
	char str[30];

public:
	void SetString(const char *s);
	void ShowString();
};

void Printer::SetString(const char* s) {
	strcpy(str, s);


}
void Printer::ShowString() {
	std::cout << str << std::endl;

}

int main(void) {
	Printer pnt;
	pnt.SetString("Hello world!");
	pnt.ShowString();

	pnt.SetString("I love C++");
	pnt.ShowString();
	return 0;
}
